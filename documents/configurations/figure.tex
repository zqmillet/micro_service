\usepackage{tikz, pgfplots}
  \usetikzlibrary{calc}
  \usetikzlibrary{fit}
  \usetikzlibrary{backgrounds}
  \usetikzlibrary{shapes}
  \usepgflibrary{arrows.meta}
  \usepgfplotslibrary{groupplots}

  \tikzset{%
    > = Stealth,
    block/.style = {rectangle, draw, minimum width = 2cm, minimum height = 1.5cm, inner sep = 0pt, align = flush center},
    node/.style = {circle, draw, minimum size = 0.8cm, inner sep = 0pt, align = flush center}
  }

  \pgfplotsset{%
    compat = 1.14,%
    filter discard warning = false,
    discard if not/.style 2 args = {
      x filter/.code = {
        \edef\tempa{\thisrow{#1}}
        \edef\tempb{#2}
        \ifx\tempa\tempb
        \else
            \def\pgfmathresult{inf}
        \fi
      }
    },
    legend cell align = {left},
    legend style = {nodes = {scale = 0.7, transform shape}},
    every tick label/.append style = {scale = 0.7},
    every axis label/.append style = {scale = 0.7},
    every axis title/.append style = {scale = 0.7},
    contour/label node code/.code={\node[sloped, scale = 0.7, fill = none] {\shadowtext{\pgfmathprintnumber{#1}}};}
  }

\usepackage{contour}
  \contourlength{1pt}
  \makeatletter
    \con@outlinetrue
  \makeatother

  \newcommand{\shadowtext}[2][white]{\contour{#1}{#2}}
